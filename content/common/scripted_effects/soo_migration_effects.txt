soo_embrace_local_culture_effect = {
    set_culture = $CULTURE$
    every_spouse = {
        limit = { 
            is_landed = no
            is_courtier_of = root 
        }
        set_culture = $CULTURE$
    }
    every_close_family_member = {
        limit = {
            NOT = { is_spouse_of = root }
            is_landed = no
            is_courtier_of = root  
        }
        set_culture = $CULTURE$
    }
    every_vassal = {
        limit = {
            is_ai = yes 
        }
        set_culture = $CULTURE$
        hidden_effect = {
            every_spouse = {
                limit = { 
                    NOT = { is_spouse_of = prev }
                    is_landed = no
                    is_courtier_of = prev 
                }
                set_culture = $CULTURE$
            }
            every_close_family_member = {
                limit = {
                    NOT = { is_spouse_of = prev }
                    is_landed = no
                    is_courtier_of = prev 
                }
                set_culture = $CULTURE$
            }
        }
    }
}
